22  * as wildcards, and their matching equivalent in the second

691  if (std::is_same<A, IntLiteral>::value) {
692  b.make_folded_const(val_b, ty, state);
693  if ((std::is_same<Op, And>::value && val_b.u.u64 == 0) ||

702  } else {

704  if ((std::is_same<Op, And>::value && val_a.u.u64 == 0) ||
705  (std::is_same<Op, Or>::value && val_a.u.u64 == 1)) {
706  // Short circuit

708  return;
709  }
710  const uint16_t l = ty.lanes;
711  b.make_folded_const(val_b, ty, state);
712  ty.lanes |= l;
713  }

715  case halide_type_int:
716  val.u.i64 = constant_fold_bin_op<Op>(ty, val_a.u.i64, val_b.u.i64);
717  break;
718  case halide_type_uint:
719  val.u.u64 = constant_fold_bin_op<Op>(ty, val_a.u.u64, val_b.u.u64);
720  break;
721  case halide_type_float:
722  case halide_type_bfloat:
723  val.u.f64 = constant_fold_bin_op<Op>(ty, val_a.u.f64, val_b.u.f64);

726  // unreachable
727  ;
728  }
